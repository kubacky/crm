import { Component, OnInit } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
import { DevicesService } from 'src/app/services/devices.service';
import { Device } from 'src/app/interfaces/device';

@Component({
  selector: 'crm-view-device',
  templateUrl: './view-device.component.html',
  styleUrls: ['./view-device.component.scss']
})
export class ViewDeviceComponent implements OnInit {

  device: Device;
  sourceId: number;

  constructor(
    private route: ActivatedRoute,
    private devicesService: DevicesService
  ) { }

  ngOnInit() {
    this.getRouteParams();
  }

  getRouteParams(): void {
    this.route.paramMap.subscribe(
      params => {
        const id = Number(params.get('id'));
        this.sourceId = id;
        this.getDevice(id);
      }
    )
  }

  private getDevice(deviceId: number): void {
    this.devicesService.getDevice(deviceId).subscribe(
      device => {
        this.device = device;
        this.setCoordinates();
      }
    )
  }

  private setCoordinates(): void {
    this.device.address.lat = Number(this.device.address.lat);
    this.device.address.lng = Number(this.device.address.lng)
  }

  showCustomerInfo(): void {
    
  }
}
